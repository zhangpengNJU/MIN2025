/**
 * Return {@code true} if the integer argument includes the
 * {@code final} modifier, {@code false} otherwise.
 *
 * @param   mod a set of modifiers
 * @return {@code true} if {@code mod} includes the
 * {@code final} modifier; {@code false} otherwise.
 */
public static boolean isFinal(int mod) {
    return (mod & FINAL) != 0;
}
