/**
 * {@inheritDoc}
 * @since 1.2
 */
public void setRoundRect(double x, double y, double w, double h, double arcw, double arch) {
    this.x = (float) x;
    this.y = (float) y;
    this.width = (float) w;
    this.height = (float) h;
    this.arcwidth = (float) arcw;
    this.archeight = (float) arch;
}
