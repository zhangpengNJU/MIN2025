/**
 * Gets the numeric era {@code int} value.
 * <p>
 * The era BEFORE_BE has the value 0, while the era BE has the value 1.
 *
 * @return the era value, from 0 (BEFORE_BE) to 1 (BE)
 */
@Override
public int getValue() {
    return ordinal();
}
