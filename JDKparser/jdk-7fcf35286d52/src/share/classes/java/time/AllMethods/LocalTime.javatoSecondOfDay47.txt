/**
 * Extracts the time as seconds of day,
 * from {@code 0} to {@code 24 * 60 * 60 - 1}.
 *
 * @return the second-of-day equivalent to this time
 */
public int toSecondOfDay() {
    int total = hour * SECONDS_PER_HOUR;
    total += minute * SECONDS_PER_MINUTE;
    total += second;
    return total;
}
