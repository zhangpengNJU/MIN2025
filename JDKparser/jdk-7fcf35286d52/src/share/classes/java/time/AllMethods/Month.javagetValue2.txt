/**
 * Gets the month-of-year {@code int} value.
 * <p>
 * The values are numbered following the ISO-8601 standard,
 * from 1 (January) to 12 (December).
 *
 * @return the month-of-year, from 1 (January) to 12 (December)
 */
public int getValue() {
    return ordinal() + 1;
}
