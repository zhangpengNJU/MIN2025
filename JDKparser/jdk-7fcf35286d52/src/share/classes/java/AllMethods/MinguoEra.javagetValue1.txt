/**
 * Gets the numeric era {@code int} value.
 * <p>
 * The era BEFORE_ROC has the value 0, while the era ROC has the value 1.
 *
 * @return the era value, from 0 (BEFORE_ROC) to 1 (ROC)
 */
@Override
public int getValue() {
    return ordinal();
}
