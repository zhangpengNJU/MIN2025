/**
 * Returns the <code>String</code> representation of this
 * <code>Rectangle2D</code>.
 * @return a <code>String</code> representing this
 * <code>Rectangle2D</code>.
 * @since 1.2
 */
public String toString() {
    return getClass().getName() + "[x=" + x + ",y=" + y + ",w=" + width + ",h=" + height + "]";
}
