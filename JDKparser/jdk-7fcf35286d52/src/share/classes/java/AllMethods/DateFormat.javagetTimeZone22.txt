/**
 * Gets the time zone.
 * This method is equivalent to the following call.
 * <blockquote><pre>{@code
 * getCalendar().getTimeZone()
 * }</pre></blockquote>
 *
 * @return the time zone associated with the calendar of DateFormat.
 */
public TimeZone getTimeZone() {
    return calendar.getTimeZone();
}
