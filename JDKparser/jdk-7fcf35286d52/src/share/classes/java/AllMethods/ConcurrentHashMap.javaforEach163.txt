public void forEach(Consumer<? super K> action) {
    if (action == null)
        throw new NullPointerException();
    Node<K, V>[] t;
    if ((t = map.table) != null) {
        Traverser<K, V> it = new Traverser<K, V>(t, t.length, 0, t.length);
        for (Node<K, V> p; (p = it.advance()) != null; ) action.accept(p.key);
    }
}
