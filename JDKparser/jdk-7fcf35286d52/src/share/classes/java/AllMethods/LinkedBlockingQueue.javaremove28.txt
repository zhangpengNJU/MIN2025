public void remove() {
    if (lastRet == null)
        throw new IllegalStateException();
    fullyLock();
    try {
        Node<E> node = lastRet;
        lastRet = null;
        for (Node<E> trail = head, p = trail.next; p != null; trail = p, p = p.next) {
            if (p == node) {
                unlink(p, trail);
                break;
            }
        }
    } finally {
        fullyUnlock();
    }
}
