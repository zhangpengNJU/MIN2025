/**
 * Returns {@code true} indicating this {@code GregorianCalendar}
 * supports week dates.
 *
 * @return {@code true} (always)
 * @see #getWeekYear()
 * @see #setWeekDate(int,int,int)
 * @see #getWeeksInWeekYear()
 * @since 1.7
 */
@Override
public final boolean isWeekDateSupported() {
    return true;
}
