private boolean nonSyncContentEquals(AbstractStringBuilder sb) {
    char[] v1 = value;
    char[] v2 = sb.getValue();
    int n = v1.length;
    if (n != sb.length()) {
        return false;
    }
    for (int i = 0; i < n; i++) {
        if (v1[i] != v2[i]) {
            return false;
        }
    }
    return true;
}
