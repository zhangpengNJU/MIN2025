/**
 * Returns the ASN.1 DER encoded form of this
 * {@code PolicyQualifierInfo}.
 *
 * @return the ASN.1 DER encoded bytes (never {@code null}).
 * Note that a copy is returned, so the data is cloned each time
 * this method is called.
 */

}
