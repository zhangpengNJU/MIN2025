private void resetState() {
    this.protocol = null;
    this.host = null;
    this.port = -1;
    this.file = null;
    this.authority = null;
    this.ref = null;
    this.hashCode = -1;
    this.handler = null;
    this.query = null;
    this.path = null;
    this.userInfo = null;
    this.tempState = null;
}
